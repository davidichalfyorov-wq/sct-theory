\begin{abstract}
We analyze the black hole information paradox within the framework of
Spectral Causal Theory (SCT), a quadratic gravity theory derived from
the Chamseddine--Connes spectral action with Standard Model content.
The SCT black hole entropy is $S = A/(4G) + 13/(120\pi) + (37/24)\ln(A/\ell_P^2)$,
where the logarithmic coefficient $c_{\log} = 37/24$ is a parameter-free
prediction from the SM field content.

We compute the SCT correction to the scrambling time:
$\delta t_s / t_s \sim 10^{-78}$ for a $10\,M_\odot$ black hole,
arising from the logarithmic entropy correction and unobservable by
any conceivable measurement.  The Page time remains
$t_{\rm Page} \sim 10^{70}$~years, with exponentially suppressed SCT
corrections of order $\exp(-m_2 r_s)$.

The fakeon prescription (Anselmi 2017) projects the ghost pole at
$z_L = 1.2807$ to zero physical degrees of freedom, giving
$\delta c_{\log} = 0$ exactly and ensuring that the bulk entanglement
entropy $S_{\rm bulk}$ receives no ghost contribution.  We identify
the Solodukhin hybrid quantum state (arXiv:2212.13208) as the mechanism
most closely aligned with the fakeon prescription, producing a Page curve
without invoking the island formula.

The SCT nonlocality scale $\ell_{\rm NL} = 1/\Lambda \sim 0.08$~mm
is insufficient to resolve the AMPS firewall for astrophysical black holes
($\ell_{\rm NL}/r_s \sim 10^{-8}$), but becomes relevant for
near-critical masses $M \sim M_{\rm crit} \sim 10^{-10}\,M_\odot$.

\section{Scrambling Time}
\label{sec:scrambling}

The scrambling time is the minimum time for a perturbation to become
``fully mixed'' with the black hole's internal degrees of freedom.
Sekino and Susskind (2008) conjectured:
\begin{equation}
\label{eq:scrambling-gr}
t_s = \frac{\beta}{2\pi}\,\ln S_{\rm BH},
\end{equation}
where $\beta = 1/T_H = 8\pi G M / c^3$ is the inverse Hawking temperature.
This was proven as a bound by Maldacena, Shenker, and Stanford (2016):
$\lambda_L \leq 2\pi / \beta$, where $\lambda_L$ is the Lyapunov exponent
of out-of-time-ordered correlators.

\subsection{SCT correction}

In SCT, the Hawking temperature receives exponentially suppressed
corrections from the ghost/fakeon sector:
\begin{equation}
T_H^{\rm SCT} = T_H^{\rm GR}\,\bigl[1 + \order(e^{-m_2 r_s})\bigr],
\end{equation}
with $m_2 = \Lambda\sqrt{60/13} \approx 2.148\,\Lambda$ and $r_s = 2GM/c^2$.
For a $10\,M_\odot$ black hole: $m_2 r_s \sim 10^8$, so the temperature
correction is $\sim e^{-10^8} \approx 0$.

The dominant SCT correction to the scrambling time comes from the entropy:
\begin{align}
t_s^{\rm SCT} &= \frac{\beta}{2\pi}\,\ln S_{\rm BH}^{\rm SCT}
               = \frac{\beta}{2\pi}\,\ln\!\left(S_{\rm GR} + \delta S\right) \\
              &= t_s^{\rm GR} + \frac{\beta}{2\pi}\,\frac{\delta S}{S_{\rm GR}}
                 + \order\!\left(\frac{\delta S}{S_{\rm GR}}\right)^{\!2},
\end{align}
where $\delta S = \alpha_C/\pi + c_{\log}\ln(A/\ell_P^2)$.

\begin{proposition}[Scrambling time correction]
\label{prop:scrambling}
For a Schwarzschild black hole of mass $M$,
\begin{equation}
\frac{\delta t_s}{t_s}
  \approx \frac{c_{\log}\,\ln(A/\ell_P^2)}{(A/4\ell_P^2)\,\ln(A/4\ell_P^2)}.
\end{equation}
For $M = 10\,M_\odot$: $\delta t_s/t_s \sim 10^{-78}$.
\end{proposition}

\begin{proof}
Direct computation.  For $M = 10\,M_\odot$:
\begin{itemize}
\item $r_s \approx 2.95 \times 10^4$~m,
\item $A = 4\pi r_s^2 \approx 1.095 \times 10^{10}$~m$^2$,
\item $A/\ell_P^2 \approx 4.19 \times 10^{79}$,
\item $S_{\rm GR} = A/(4\ell_P^2) \approx 1.05 \times 10^{79}$,
\item $\delta S = 13/(120\pi) + (37/24)\ln(4.19 \times 10^{79})
       \approx 0.035 + 1.542 \times 183.5 \approx 283$,
\item $\delta t_s/t_s \approx 283 / (1.05 \times 10^{79} \times 182)
       \approx 1.5 \times 10^{-79}$.
\end{itemize}
\end{proof}

\verified{Numerical: \texttt{lt2\_information.py},
  \texttt{verify\_scrambling\_10msun()}: $\log_{10}(\delta t/t) \in [-80, -75]$.
  VR-LT2-001.}

\subsection{Fakeon threshold in scrambling dynamics}

The result $\delta t_s/t_s \sim 10^{-78}$ assumes that the Lyapunov
exponent $\lambda_L = 2\pi/\beta$ is unchanged by the fakeon sector.
However, the MSS chaos bound relies on unitarity, analyticity, and
\emph{causality} of the OTOC (Maldacena--Shenker--Stanford 2016).
The fakeon prescription violates microcausality (Anselmi--Piva 2018),
so the MSS prerequisites are not guaranteed.

In the MSS shock-wave picture, the center-of-mass energy squared of
near-horizon scattering grows as
\begin{equation}
s \sim (k_B T_H)^2 \exp\!\left(\frac{2\pi t}{\beta}\right).
\end{equation}
The fakeon threshold $\sqrt{s} = m_2$ is reached at
\begin{equation}
\label{eq:t-chi}
t_\chi = \frac{\beta}{\pi}\ln\!\left(\frac{m_2}{k_B T_H}\right).
\end{equation}
For $M = 10\,M_\odot$:
\begin{itemize}
\item $m_2/T_H \approx 9.62 \times 10^9$,
\item $\ln(m_2/T_H) \approx 23.0$,
\item $t_\chi \approx 9.1$~ms,
\item $t_s^{\rm GR} \approx 35.9$~ms,
\item $t_\chi / t_s \approx 0.25$.
\end{itemize}

\begin{proposition}[Fakeon threshold precedes scrambling]
\label{prop:t-chi}

is less than the scrambling time $t_s$:
\begin{equation}

                   = \frac{2\ln(m_2/k_B T_H)}{\ln[A/(4\ell_P^2)]}
                   \approx 0.25 \quad (10\,M_\odot).
\end{equation}
The fakeon sector enters the chaos dynamics at one quarter of the
scrambling time.
\end{proposition}

\section{Page Time and Page Curve}
\label{sec:page}

\subsection{Page time}

The Page time is the timescale at which the entanglement entropy of
Hawking radiation reaches its maximum.  For Schwarzschild evaporation
(massless emission, Page 1976):
\begin{equation}
t_{\rm evap} = \frac{5120\,\pi\,G^2 M^3}{\hbar\,c^4},
\qquad
t_{\rm Page} = \frac{t_{\rm evap}}{2}.
\end{equation}

For $M = 10\,M_\odot$: $t_{\rm Page} \sim 10^{70}$~years.

\begin{equation}
\frac{\delta t_{\rm Page}}{t_{\rm Page}}
  = \order\!\left(e^{-m_2 r_s}\right)
  \sim e^{-10^8} \approx 0.

\subsection{Schematic Page curve}

The Page curve has two branches:
\begin{equation}

\begin{cases}
S_{\rm BH}^0 \cdot 2t/t_{\rm evap} & t < t_{\rm Page}, \\
S_{\rm BH}^0 \cdot (1 - t/t_{\rm evap})^{2/3} & t > t_{\rm Page}.

The early branch reflects thermodynamic entropy production; the late
branch reflects the shrinking Bekenstein--Hawking entropy of the

\subsection{Blueshift argument: SCT does touch the pair-creation region}

A late Hawking quantum has asymptotic energy $E_\infty \sim k_B T_H$,

\subsection{Mathur's small corrections theorem}

Mathur (2009, arXiv:0909.1038) proved that if corrections to the
Hawking emission process are perturbatively small (of order~$\epsilon$),
then the entanglement entropy of the radiation deviates from the
thermal result by $\order(\epsilon)$.  To recover the Page curve,
one needs $\order(1)$ corrections to the near-horizon state.

Corley and Jacobson (1996, arXiv:hep-th/9601073) showed that UV
dispersion modifications (analogous to form-factor nonlocality)
preserving local Lorentz invariance leave the Hawking spectrum
approximately thermal.  This suggests that the form-factor
modification in the $\rho \lesssim \rho_\Lambda$ layer is
a \emph{small} correction to the Hawking state.

\subsection{Comparison with Giddings}

at scale $\ell_{\rm NL}^{\rm Giddings} \sim r_s$ (maximal nonlocality).
SCT provides $\ell_{\rm NL} = 1/\Lambda \ll r_s$ (minimal nonlocality).

\begin{proposition}[Firewall status in SCT]
\label{prop:firewall}
SCT's form factors modify the trans-$\Lambda$ precursor region of
Hawking quanta in a microscopic proper-distance layer
$\rho \lesssim \rho_\Lambda \approx 13\;\mu$m from the horizon.
However, for astrophysical black holes ($M \gg M_{\rm crit}$), there
is no shown SCT mechanism that converts this microscopic UV modification
into an $\order(1)$, state-sensitive restructuring of entanglement
after the Page time.  The firewall paradox is not demonstrably resolved
by SCT nonlocality alone.
\end{proposition}

\section{Island Formula with Nonlocal Action}
\label{sec:island}

\subsection{Wald--Dong entropy for SCT}

The generalized entropy functional for the island formula in SCT is
\begin{equation}
\label{eq:gen-entropy}
S_{\rm gen} = S_{\rm WD}(\partial I) + S_{\rm bulk}(R \cup I),
\end{equation}
where the Wald--Dong entropy is (Dong 2014):
\begin{equation}
\label{eq:wald-dong}
S_{\rm WD} = \frac{A}{4G}
  + \frac{\alpha_C}{\pi}
  + c_{\log}\ln\!\left(\frac{A}{\ell_P^2}\right)
  - \frac{\alpha_C}{\pi}\int_{\partial I}
    \sigma^{(k)}\sigma^{(l)}\,\dd A.
\end{equation}

The last term is the Dong anomaly correction, proportional to the
extrinsic curvature of the entangling surface $\partial I$.  On a
bifurcation surface (stationary BH), $\sigma^{(k)} = 0$ and the
correction vanishes.

\subsection{Replica entropy vs.\ running $G_{\rm eff}$}

\caveat{The propagator running coupling
  $G_{\rm eff}(k) = G/\PiTT(k^2/\Lambda^2)$ should \textbf{not}
  be substituted into the island formula's area term.  The correct
  gravitational entropy in the island formula is the replica entropy
  $S_{\rm grav}^{\rm rep,SCT}[\partial I]$ of the full nonlocal action,
  not $A/(4G_{\rm eff})$ (Dong 2014, Camps 2014).  In the local
  (long-wavelength) limit, this reduces to the Wald--Dong functional
  $S_{\rm WD}$.}

For entire form factors $F(\Box/\Lambda^2)$, no closed-form ``Dong
formula in one line'' exists.  The standard approach is either:
(a)~expand $F = \sum_n f_n(\Box/\Lambda^2)^n$ and compute the
conical entropy term by term; or (b)~localize via auxiliary fields.
The \emph{splitting problem} (Miao--Guo 2014, arXiv:1411.5579) is open.

\subsection{Nonlocal corrections and QES shift}

The EFT expansion parameter at the island surface is
\begin{equation}
z_{\rm island} = \left(\frac{\ell_{\rm NL}}{r_s}\right)^2
  \approx 7.9 \times 10^{-18} \quad (10\,M_\odot),
\end{equation}
so nonlocal corrections are suppressed by $(\ell_{\rm NL}/r_s)^2$
relative to local terms.  The QES shift is
\begin{equation}
\delta r_{\rm QES} \sim \frac{\ell_{\rm NL}^2}{r_s}
  \approx 2.3 \times 10^{-13}\;\text{m},
\qquad
\frac{\delta r_{\rm QES}}{r_s} \sim 10^{-17},
\end{equation}
not $\sim \ell_{\rm NL} \sim 10^{-4}$~m.  The quadratic scaling
arises because the form-factor expansion parameter is
$\Box/\Lambda^2 \sim 1/(\Lambda r_s)^2$, not $1/(\Lambda r_s)$.

The local SCT correction to the stationary entropy remains
\begin{equation}
\frac{\delta S_{\rm local}}{S_{\rm BH}} \sim 10^{-77},
\end{equation}
and the additional nonlocal correction is suppressed by $z_{\rm island}
\sim 10^{-18}$, giving a total correction at the $10^{-77}$ level.

\section{Solodukhin Hybrid Mechanism}
\label{sec:solodukhin}

Potaux, Sarkar, and Solodukhin (2022, arXiv:2212.13208) proposed a
``hybrid quantum state'' for theories with ghost-like modes:
\begin{itemize}
\item Physical particles: Hartle--Hawking state (thermal).
\item Ghost modes: Boulware state (vacuum).
\end{itemize}

In the RST model, this hybrid state produces:
\begin{enumerate}
\item A regular causal diamond spacetime (no singularity).
\item A Page curve for the entropy of asymptotic radiation.
\item Information recovery through correlations between physical
      and ghost virtual pairs.
\end{enumerate}

Crucially, in a follow-up paper (2024, arXiv:2411.09574), they showed
that the Page curve arises \emph{without} the island mechanism.  The
island does not contribute to the generalized entropy in this setup.

\begin{proposition}[Solodukhin mechanism: analogy, not equivalence]
The Solodukhin hybrid quantum state is the \emph{closest existing
analogy} to SCT's fakeon prescription, but the two are \textbf{not
equivalent}.
\begin{itemize}
\item \textbf{Hybrid state (Solodukhin):} The non-physical field
  remains in the semiclassical dynamics as a separate sector placed
  in the Boulware state.  Physical/non-physical correlations produce
  the Page curve.  In the 2024 paper, non-physical partners later
  ``re-emerge and carry missing information.''
\item \textbf{Fakeon (Anselmi):} The would-be ghost is ``virtualized
  completely and eradicated from the theory'' --- it has no physical
  Hilbert-space sector at all.  There is no separate state assignment
  (Boulware or otherwise) for the fakeon.
\end{itemize}
Both mechanisms remove the thermal ghost sector, but the fakeon
prescription is \emph{strictly stronger}: it projects out the ghost
entirely, rather than assigning it a particular quantum state.
\end{proposition}

\caveat{The Solodukhin mechanism has been demonstrated only in 2D
  toy models (RST, CGHS).  Extension to 4D is open.
  The fakeon $\neq$ Boulware ghost: transferring Solodukhin's
  Page-curve result to SCT requires a separate derivation,
  not a direct mapping.}

\section{Regge Kernel Theorem: $\lambda_L = 2\pi T_H$}
\label{sec:regge}

The preceding sections established $\lambda_L = 2\pi T_H$ through
two methods: eikonal propagator analysis and the fakeon shock-wave
profile.  We now present a \textbf{comprehensive proof} using four
independent methods and identify the new prediction $v_B^{\rm SCT}$.

\subsection{Method 1: MR-7 field redefinition (tree level, exact)}

MR-7 (CERTIFIED, 64/64 tests) proved the Modesto--Calcagni theorem:
\begin{equation}
\label{eq:m-tree}
\mathcal{M}_{\rm tree}^{\rm SCT}(s,t)
= \mathcal{M}_{\rm tree}^{\rm GR}(s,t)
\end{equation}
for on-shell external states on Ricci-flat backgrounds.  The conditions
(E$\cdot$F$\cdot$E action structure, $F_{1,2}$ entire, background
Ricci-flat) are all verified.

\begin{theorem}[$\lambda_L$ at tree level]
\label{thm:lyapunov}
At tree level, $\lambda_L^{\rm SCT} = 2\pi T_H$ exactly.
\end{theorem}

\begin{proof}
The eikonal phase is a functional of the on-shell amplitude:
$\delta(s,b) = (1/2s)\int \dd^2 q\, e^{-iq\cdot b}\,
\mathcal{M}(s,t{=}{-}q^2)/(2\pi)^2$.
Since $\mathcal{M}_{\rm tree}^{\rm SCT} = \mathcal{M}_{\rm tree}^{\rm GR}$
(\cref{eq:m-tree}), $\delta_{\rm SCT} = \delta_{\rm GR}$ at tree level.
The Lyapunov exponent is extracted from $\delta \propto e^{\lambda_L t}$,
giving $\lambda_L = 2\pi T_H$.
\end{proof}

\subsection{Method 2: Vertex momentum routing (explicit mechanism)}

At cubic level in the action $\int C\,F_1(\Box/\Lambda^2)\,C$:
\begin{equation}
F_1(\Box)\,C^{(2)}(k_j,k_k) = F_1(-(k_j{+}k_k)^2/\Lambda^2)\,C^{(2)}.
\end{equation}
By momentum conservation: $k_j + k_k = -k_i$.  Therefore
$F_1 = F_1(-k_i^2/\Lambda^2)$.  For on-shell external legs ($k_i^2=0$):
$F_1(0) = 1$.  For the exchange ($k_i^2 = t$): $F_1(-t/\Lambda^2)$.
No $s$-dependence $\Rightarrow$ Regge intercept $j(0) = 2$.

\subsection{Method 3: Pole-skipping}

At the first Matsubara frequency $\omega_1 = i\cdot 2\pi T_H$, the
near-horizon retarded Green's function has a degenerate point
(simultaneous pole and zero).  In SCT, the form factor modifies
$k_*$ but not $\omega_*$:
\begin{equation}
\omega_* = i\cdot 2\pi T_H \quad\text{(unchanged)}, \qquad
k_*^{\rm SCT} \neq k_*^{\rm GR}.
\end{equation}
The correction to $k_*$ is of order $(T_H/\Lambda)^2 \sim 10^{-20}$
for $10\,M_\odot$.  This gives $\lambda_L = 2\pi T_H$ and
$v_B^{\rm SCT} = v_B^{\rm GR} + \order(10^{-20})$.

\subsection{Method 4: Froissart / spin content}

SCT contains only spin-0 and spin-2 exchanges.  The entire form
factors dress vertices by scalar functions of momenta, not higher-spin
tensors.  Therefore $j(0) = \max(\text{spin}) = 2$, and $\lambda_L
= 2\pi T_H$.

      arising from the logarithmic entropy correction $c_{\log} = 37/24$.
      This is unobservable by any conceivable measurement.

      the ghost contributes nothing to the entanglement entropy.
      Consequently, $S_{\rm bulk}^{\rm SCT} = S_{\rm bulk}^{\rm GR}$

      correction, which modifies the island surface position at
      order $\sim 10^{-76}$.
